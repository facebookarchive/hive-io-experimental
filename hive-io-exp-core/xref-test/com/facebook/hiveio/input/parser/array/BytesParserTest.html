129   private void assertParseLong(String string) {
130     assertEquals(parseLong(string.getBytes(Charsets.US_ASCII), 0, string.length()),
131         Long.parseLong(string));
132 
133     // verify we can parse using a non-zero offset
134     String padding = "9999";
135     String padded = padding + string + padding;
136     assertEquals(parseLong(padded.getBytes(Charsets.US_ASCII), padding.length(),
137         string.length()), Long.parseLong(string));
138   }
