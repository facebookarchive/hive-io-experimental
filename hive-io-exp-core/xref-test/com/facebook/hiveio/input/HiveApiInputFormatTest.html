46  public class HiveApiInputFormatTest {
47    private final File ROOT_DIR = new File("/tmp/hive-io-test");
48  
49    private HiveConf hiveConf;
50    private HiveInterface client;
51  
52    private File metastoreDir() {
53      return new File(ROOT_DIR, "metastore_db");
54    }
55  
56    @BeforeSuite
57    public void beforeSuite() throws Exception {
58      HadoopNative.requireHadoopNative();
59      ROOT_DIR.mkdirs();
60      FileUtils.deleteDirectory(ROOT_DIR);
61      hiveConf = new HiveConf(HiveApiInputFormatTest.class);
62      hiveConf.setVar(HiveConf.ConfVars.METASTOREWAREHOUSE,
63          metastoreDir().getAbsolutePath());
64      hiveConf.setVar(HiveConf.ConfVars.METASTORECONNECTURLKEY,
65          "jdbc:derby:;databaseName=" + metastoreDir().toString() + ";create=true");
66      client = new HiveServer.HiveServerHandler(hiveConf);
67    }
68  
69    @Test
70    public void testInput() throws Exception {
71      String tableName = "t1";
72      initData(tableName);
73      run1(tableName);
74    }
75  
76    private void run1(String tableName) throws IOException, InterruptedException {
77      HiveInputDescription hid = new HiveInputDescription();
78      hid.setTableName(tableName);
79      HiveApiInputFormat.setProfileInputDesc(hiveConf, hid,
80          HiveApiInputFormat.DEFAULT_PROFILE_ID);
81  
82      HiveApiInputFormat haif = new HiveApiInputFormat();
83  
84      List<InputSplit> splits = haif.getSplits(hiveConf, client);
85  
86      TaskAttemptContext taskContext = new TaskAttemptContext(hiveConf, new TaskAttemptID());
87      InputSplit split = splits.get(0);
88      RecordReader<WritableComparable, HiveReadableRecord> recordReader =
89          haif.createRecordReader(split, taskContext);
90      recordReader.initialize(split, taskContext);
91  
92      assertTrue(recordReader.nextKeyValue());
93      HiveReadableRecord record = recordReader.getCurrentValue();
94      System.out.println(record);
95      assertEquals(Long.class, record.get(0).getClass());
96      assertEquals(Double.class, record.get(1).getClass());
97      assertEquals(1, record.getLong(0));
98      assertEquals(1.1, record.getDouble(1));
99  
100     assertTrue(recordReader.nextKeyValue());
101     record = recordReader.getCurrentValue();
102     assertEquals(2, record.getLong(0));
103     assertEquals(2.2, record.getDouble(1));
104 
105     assertFalse(recordReader.nextKeyValue());
106   }

108   private void initData(String tableName)

115       "",
116     };
117     File dataFile = new File(ROOT_DIR, "foo");

119     client.execute("" +
120         "CREATE TABLE " + tableName +
121         " (i1 INT, d1 DOUBLE) " +
122         " ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t'");
123     client.execute("LOAD DATA LOCAL INPATH '" + dataFile.getAbsolutePath() +
124         "' INTO TABLE " + tableName);
125 
126     System.out.println("FINISHED LOADING DATA");
127   }
