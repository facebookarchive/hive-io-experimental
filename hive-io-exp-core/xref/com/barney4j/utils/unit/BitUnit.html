115   static final double C_KIBIT = Math.pow(2d, 10d);
116   static final double C_MIBIT = Math.pow(2d, 20d);
117   static final double C_GIBIT = Math.pow(2d, 30d);
118   static final double C_TIBIT = Math.pow(2d, 40d);
119   static final double C_PIBIT = Math.pow(2d, 50d);

122   static final double C_MBIT = Math.pow(10d, 6d);
123   static final double C_GBIT = Math.pow(10d, 9d);
124   static final double C_TBIT = Math.pow(10d, 12d);
125   static final double C_PBIT = Math.pow(10d, 15d);

128   private static final double MAX = Double.MAX_VALUE;
129 
130   static final double safeMulti(double d, double multi) {
131     double limit = MAX / multi;
132 
133     if (d >  limit) {
134       return Double.MAX_VALUE;
135     }
136     
137     if (d < -limit) {
138       return Double.MIN_VALUE;
139     }
140 
141     return d * multi;
142   }
