18  package com.facebook.giraph.hive.common;
19  
20  import org.apache.hadoop.hive.ql.stats.StatsSetupConst;
21  
22  import com.barney4j.utils.unit.ByteUnit;
23  import com.google.common.base.Objects;
24  
25  import java.util.Map;
26  
27  import static java.lang.Long.parseLong;
28  
29  /**
30   * Information about a table or partition. See "DESC FORMATTED table"
31   */
32  public class HiveStats {
33    private long totalSize;
34    private long rawSize;
35    private long numRows;
36  
37    public HiveStats() {
38      totalSize = 0;
39      rawSize = 0;
40      numRows = 0;
41    }
42  
43    public HiveStats(long numRows, long rawSize, long totalSize) {
44      this.numRows = numRows;
45      this.rawSize = rawSize;
46      this.totalSize = totalSize;
47    }
48  
49    public long getNumRows() {
50      return numRows;
51    }
52  
53    public long getRawSizeInBytes() {
54      return rawSize;
55    }
56  
57    public double getRawSizeInMB() {
58      return ByteUnit.BYTE.toMB(rawSize);
59    }
60  
61    public long getTotalSizeInBytes() {
62      return totalSize;
63    }
64  
65    public double getTotalSizeInMB() {
66      return ByteUnit.BYTE.toMB(totalSize);
67    }
68  
69    public void add(HiveStats other) {
70      totalSize += other.totalSize;
71      rawSize += other.rawSize;
72      numRows += other.numRows;
73    }
74  
75    public static HiveStats fromParams(Map<String, String> params) {
76      long totalSize = parseLong(params.get(StatsSetupConst.TOTAL_SIZE));
77      long rawSize = parseLong(params.get(StatsSetupConst.RAW_DATA_SIZE));
78      long numRows = parseLong(params.get(StatsSetupConst.ROW_COUNT));
79      return new HiveStats(numRows, rawSize, totalSize);
80    }
81  
82    @Override public String toString() {
83      return Objects.toStringHelper(this)
84          .add("totalSize", totalSize)
85          .add("rawSize", rawSize)
86          .add("numRows", numRows)
87          .toString();
88    }
89  }
