42  /**
43   * OutputCommitter for Hive output
44   */
45  class HiveApiOutputCommitter extends OutputCommitter {
46    /** Base Hadoop output committer */
47    private final OutputCommitter baseCommitter;
48    /** Profile ID to use */
49    private final String profileId;
50  
51    /**
52     * Constructor
53     *
54     * @param baseCommitter Base Hadoop committer
55     * @param profileId Profile ID
56     */
57    public HiveApiOutputCommitter(OutputCommitter baseCommitter,
58                                  String profileId) {
59      this.baseCommitter = baseCommitter;
60      this.profileId = profileId;
61    }

63    @Override
64    public void setupJob(JobContext jobContext) throws IOException {
65      baseCommitter.setupJob(jobContext);
66    }
67  
68    @Override
69    public void commitJob(JobContext jobContext) throws IOException {
70      baseCommitter.commitJob(jobContext);
71  
72      Configuration conf = jobContext.getConfiguration();
73      OutputConf outputConf = new OutputConf(conf, profileId);
74      HiveOutputDescription outputDesc = outputConf.readOutputDescription();
75      OutputInfo outputInfo = outputConf.readOutputTableInfo();
76      if (outputInfo.hasPartitionInfo()) {
77        registerPartitions(conf, outputDesc, outputInfo);
78      } else {
79        noPartitionsCopyData(conf, outputInfo);
80      }
81  
82      writeSuccessFile(conf);
83    }
84  
85    /**
86     * Write success file to Hadoop if required
87     *
88     * @param conf Configuration
89     * @throws IOException I/O errors
90     */
91    private static void writeSuccessFile(Configuration conf) throws IOException {
92      if (!HadoopUtils.needSuccessMarker(conf)) {
93        return;
94      }
95      Path outputPath = HadoopUtils.getOutputPath(conf);
96      FileSystem fs = FileSystem.get(outputPath.toUri(), conf);
97      if (fs.exists(outputPath)) {
98        Path successPath = new Path(outputPath, "_SUCCESS");
99        if (!fs.exists(successPath)) {
100         fs.create(successPath).close();
101       }
102     }
103   }
104 
105   /**
106    * Register partitions for new data we wrote.
107    *
108    * @param conf Configuration
109    * @param outputDesc Output description from user
110    * @param outputInfo Internal output information
111    * @throws IOException
112    */
113   private void registerPartitions(Configuration conf,
114     HiveOutputDescription outputDesc, OutputInfo outputInfo)
115     throws IOException {
116     HiveConf hiveConf = new HiveConf(conf, HiveApiOutputCommitter.class);
117 
118     String dbName = outputDesc.getDbName();
119     String tableName = outputDesc.getTableName();
120 
121     HiveMetaStoreClient client;
122     Table hiveTable;
123     try {
124       client = new HiveMetaStoreClient(hiveConf);
125       hiveTable = client.getTable(dbName, tableName);
126       // CHECKSTYLE: stop IllegalCatch
127     } catch (Exception e) {
128       // CHECKSTYLE: resume IllegalCatch
129       throw new IOException(e);
130     }
131 
132     Partition partition = new Partition();
133     partition.setDbName(dbName);
134     partition.setTableName(tableName);
135     partition.setParameters(outputInfo.getTableParams());
136     List<String> partitionValues = HiveUtils
137         .orderedPartitionValues(hiveTable.getPartitionKeys(),
138             outputDesc.getPartitionValues());
139     partition.setValues(partitionValues);
140 
141     StorageDescriptor sd = new StorageDescriptor(hiveTable.getSd());
142     sd.setParameters(outputInfo.getSerializerParams());
143     sd.setLocation(outputInfo.getFinalOutputPath());
144     sd.setCols(outputInfo.getColumnInfo());
145     partition.setSd(sd);
146 
147     try {
148       client.add_partition(partition);
149       // CHECKSTYLE: stop IllegalCatch
150     } catch (Exception e) {
151       // CHECKSTYLE: resume IllegalCatch
152       throw new IOException(e);
153     }
154   }
155 
156   /**
157    * Table has no partitions, just copy data
158    *
159    * @param conf Configuration
160    * @param oti OutputInfo
161    * @throws IOException I/O errors
162    */
163   private void noPartitionsCopyData(Configuration conf, OutputInfo oti)
164     throws IOException {
165     Preconditions.checkArgument(!oti.hasPartitionInfo());
166     Path tablePath = new Path(oti.getTableRoot());
167     Path writePath = new Path(oti.getPartitionPath());
168     FileSystem fs = FileSystem.get(tablePath.toUri(), conf);
169     FileSystems.move(fs, writePath, writePath, tablePath);
170     fs.delete(writePath, true);
171   }
