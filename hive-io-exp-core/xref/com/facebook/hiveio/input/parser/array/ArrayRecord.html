18  package com.facebook.hiveio.input.parser.array;

20  import com.facebook.hiveio.common.HiveType;
21  import com.facebook.hiveio.common.NativeType;
22  import com.facebook.hiveio.record.HiveReadableRecord;
23  import com.google.common.base.Objects;

25  import java.util.Arrays;

27  class ArrayRecord implements HiveReadableRecord {
28    private final int numColumns;
29  
30    // Note that partition data and column data are stored together, with column
31    // data coming before partition values.
32    private final HiveType[] hiveTypes;
33    private final boolean[] booleans;
34    private final long[] longs;
35    private final double[] doubles;
36    private final String[] strings;
37    private final Object[] objects;
38    private final boolean[] nulls;
39  
40    public ArrayRecord(int numColumns, String[] partitionValues, HiveType[] hiveTypes) {
41      this.numColumns = numColumns;
42      this.hiveTypes = hiveTypes;
43  
44      final int size = numColumns + partitionValues.length;
45      this.booleans = new boolean[size];
46      this.longs = new long[size];
47      this.doubles = new double[size];
48      this.strings = new String[size];
49      this.objects = new Object[size];
50      this.nulls = new boolean[size];
51  
52      for (int partIndex = 0; partIndex < partitionValues.length; ++partIndex) {
53        strings[partIndex + numColumns] = partitionValues[partIndex];
54      }
55    }
56  
57    public int getNumColumns() {
58      return numColumns;
59    }
60  
61    public int getNumPartitionValues() {
62      return booleans.length - numColumns;
63    }
64  
65    public int getSize() {
66      return booleans.length;
67    }
68  
69    public void reset() {
70      Arrays.fill(nulls, false);
71    }
72  
73    public HiveType getHiveType(int index) {
74      return hiveTypes[index];
75    }
76  
77    public NativeType getNativeType(int index) {
78      return hiveTypes[index].getNativeType();
79    }
80  
81    public void setBoolean(int index, boolean value) {
82      booleans[index] = value;
83    }
84  
85    public void setLong(int index, long value) {
86      longs[index] = value;
87    }
88  
89    public void setDouble(int index, double value) {
90      doubles[index] = value;
91    }
92  
93    public void setString(int index, String value) {
94      strings[index] = value;
95    }
96  
97    public void setObject(int index, Object value) {
98      objects[index] = value;
99    }
100 
101   public void setNull(int index, boolean value) {
102     nulls[index] = value;
103   }
104 
105   @Override
106   public Object get(int index) {
107     if (nulls[index]) {
108       return null;
109     }
110     if (hiveTypes[index].isCollection()) {
111       return objects[index];
112     } else {
113       return getPrimitive(index);
114     }
115   }
116 
117   private Object getPrimitive(int index) {
118     switch (hiveTypes[index].getNativeType()) {
119       case BOOLEAN:
120         return booleans[index];
121       case LONG:
122         return longs[index];
123       case DOUBLE:
124         return doubles[index];
125       case STRING:
126         return strings[index];
127     }
128     return null;
129   }
130 
131   @Override
132   public boolean getBoolean(int index) {
133 //    verifyType(index, NativeType.BOOLEAN);
134     return booleans[index];
135   }
136 
137   @Override
138   public long getLong(int index) {
139 //    verifyType(index, NativeType.LONG);
140     return longs[index];
141   }
142 
143   @Override
144   public double getDouble(int index) {
145 //    verifyType(index, NativeType.DOUBLE);
146     return doubles[index];
147   }
148 
149   @Override
150   public String getString(int index) {
151 //    verifyType(index, NativeType.STRING);
152     return strings[index];
153   }
154 
155   @Override
156   public boolean isNull(int index) {
157     return nulls[index];
158   }

160   private void verifyType(int index, NativeType expectedType) {
161     if (hiveTypes[index].getNativeType() != expectedType) {
162       throw new IllegalStateException(
163           String.format("Got an unexpected type %s from row %s for column %d, should be %s",
164               hiveTypes[index], this, index, expectedType));
165     }
166   }
167 
168   @Override
169   public String toString() {
170     Objects.ToStringHelper tsh = Objects.toStringHelper(this);
171     for (int i = 0; i < getSize(); ++i) {
172       tsh.add("type[" + i + "]", hiveTypes[i]);
173       tsh.add("data[" + i + "]", get(i));
174     }
175     return tsh.toString();
176   }
