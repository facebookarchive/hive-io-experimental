48  /**
49   * InputSplit for Hive
50   */
51  class HInputSplit extends InputSplit
52      implements org.apache.hadoop.mapred.InputSplit, Configurable {
53    /** Logger */
54    private static final Logger LOG = LoggerFactory.getLogger(HInputSplit.class);
55  
56    /** Hadoop InputFormat for reading records */
57    private org.apache.hadoop.mapred.InputFormat baseInputFormat;
58    /** Hadoop InputSplit */
59    private org.apache.hadoop.mapred.InputSplit baseSplit;
60    /** Schema of table */
61    private final HiveTableSchema tableSchema;
62    /** Data for this input split */
63    private final InputSplitData inputSplitData;
64    /** Which columns to read */
65    private int[] columnIds;
66  
67    // These members are not serialized.
68    /** Hadoop Configuration */
69    private Configuration conf;
70    /** Hive Deserializer */
71    private Deserializer deserializer;
72  
73    /**
74     * Constructor for reflection
75     */
76    public HInputSplit() {
77      tableSchema = new HiveTableSchemaImpl();
78      inputSplitData = new InputSplitData();
79      columnIds = new int[0];
80    }
81  
82    /**
83     * Constructor
84     *
85     * @param baseInputFormat Hadoop InputFormat
86     * @param baseSplit Hadoop InputSplit
87     * @param tableSchema Schema for Hive table
88     * @param columnIds List of column ids
89     * @param inputSplitData Data for this split
90     * @param conf Configuration
91     */
92    public HInputSplit(org.apache.hadoop.mapred.InputFormat baseInputFormat,
93                       org.apache.hadoop.mapred.InputSplit baseSplit,
94                       HiveTableSchema tableSchema, int[] columnIds,
95                       InputSplitData inputSplitData, Configuration conf) {
96      this.baseSplit = baseSplit;
97      this.baseInputFormat = baseInputFormat;
98      this.tableSchema = tableSchema;
99      this.columnIds = columnIds;
100     this.inputSplitData = inputSplitData;
101     setConf(conf);
102   }
103 
104   public Deserializer getDeserializer() {
105     return Preconditions.checkNotNull(deserializer);
106   }
107 
108   public String[] getPartitionValues() {
109     List<String> partValues = inputSplitData.getPartitionValues();
110     return partValues.toArray(new String[partValues.size()]);
111   }
112 
113   public HiveTableSchema getTableSchema() {
114     return tableSchema;
115   }
116 
117   public int[] getColumnIds() {
118     return columnIds;
119   }
120 
121   /**
122    * GEt the Hadoop RecordReader which we will wrap with ours.
123    *
124    * @param jobConf Hadoop Job configuration
125    * @param progressable Object tracking progress
126    * @return underlying RecordReader
127    * @throws IOException I/O errors
128    */
129   public org.apache.hadoop.mapred.RecordReader<WritableComparable, Writable>
130   getBaseRecordReader(JobConf jobConf, Progressable progressable)
131     throws IOException {
132     Reporter reporter = new ProgressReporter(progressable);
133     return baseInputFormat.getRecordReader(baseSplit, jobConf, reporter);
134   }
135 
136   /**
137    * Initialize Deserializer used
138    */
139   private void initDeserializer() {
140     if (deserializer == null && inputSplitData.hasDeserializerClass()) {
141       deserializer = inputSplitData.createDeserializer();
142     }
143     if (deserializer != null && conf != null) {
144       SerDes.initDeserializer(deserializer, conf,
145           inputSplitData.getColumnInfo(),
146           inputSplitData.getDeserializerParams());
147     }
148   }
149 
150   @Override
151   public long getLength() throws IOException {
152     return baseSplit.getLength();
153   }
154 
155   @Override
156   public String[] getLocations() throws IOException {
157     return baseSplit.getLocations();
158   }
159 
160   @Override
161   public void setConf(Configuration conf) {
162     this.conf = conf;
163     HadoopUtils.setConfIfPossible(baseInputFormat, conf);
164     HadoopUtils.setConfIfPossible(baseSplit, conf);
165     HadoopUtils.setConfIfPossible(deserializer, conf);
166     initDeserializer();
167   }
168 
169   @Override
170   public Configuration getConf() {
171     if (conf == null) {
172       getConfFromMembers();
173     }
174     return conf;
175   }
176 
177   private void getConfFromMembers() {
178     Object[] objectsToCheck = new Object[] {
179       baseSplit,
180       baseInputFormat,
181       deserializer
182     };
183     for (int i = 0; conf == null && i < objectsToCheck.length; ++i) {
184       conf = HadoopUtils.getConfIfPossible(objectsToCheck[i]);
185     }
186   }
187 
188   @Override
189   public void write(DataOutput out) throws IOException {
190     Writables.writeUnknownWritable(out, baseSplit);
191     Writables.writeClassName(out, baseInputFormat);
192     tableSchema.write(out);
193     inputSplitData.write(out);
194     Writables.writeIntArray(out, columnIds);
195   }
196 
197   @Override
198   public void readFields(DataInput in) throws IOException {
199     baseSplit = Writables.readUnknownWritable(in);
200     baseInputFormat = Writables.readNewInstance(in);
201     tableSchema.readFields(in);
202     inputSplitData.readFields(in);
203     columnIds = Writables.readIntArray(in);
204     initDeserializer();
205   }
206 
207   @Override
208   public String toString() {
209     return Objects.toStringHelper(this)
210         .add("baseInputFormat class", baseInputFormat.getClass())
211         .add("tableSchema", tableSchema)
212         .add("inputSplitData", inputSplitData)
213         .add("columnIds", Arrays.toString(columnIds))
214         .add("baseSplitClass", baseSplit.getClass())
215         .add("baseSplit", baseSplit)
216         .toString();
217   }
