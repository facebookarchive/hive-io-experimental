124   /**
125    * Initialize passed in profile ID with Configuration and output description
126    * passed in.
127    * @param conf Configuration to use
128    * @param outputDesc HiveOutputDescription
129    * @param profileId Profile to use
130    * @throws TException Hive Metastore issues
131    */
132   public static void initProfile(Configuration conf,
133                                  HiveOutputDescription outputDesc,
134                                  String profileId)
135     throws TException {
136     String dbName = outputDesc.getDbName();
137     String tableName = outputDesc.getTableName();
138 
139     HiveConf hiveConf = new HiveConf(conf, HiveApiInputFormat.class);
140     HiveMetaStoreClient client = new HiveMetaStoreClient(hiveConf);
141 
142     Table table = client.getTable(dbName, tableName);
143     sanityCheck(table, outputDesc);
144 
145     OutputInfo oti = new OutputInfo(table);
146 
147     String partitionPiece;
148     if (oti.hasPartitionInfo()) {
149       partitionPiece = HiveUtils.computePartitionPath(oti.getPartitionInfo(),
150           outputDesc.getPartitionValues());
151     } else {
152       partitionPiece = "_temp";
153     }
154     String partitionPath = oti.getTableRoot() + Path.SEPARATOR + partitionPiece;
155 
156     oti.setPartitionPath(partitionPath);
157     HadoopUtils.setOutputDir(conf, partitionPath);
158 
159     if (oti.hasPartitionInfo()) {
160       oti.setFinalOutputPath(oti.getPartitionPath());
161     } else {
162       oti.setFinalOutputPath(oti.getTableRoot());
163     }
164 
165     HiveTableSchema tableSchema = HiveTableSchemaImpl.fromTable(table);
166     HiveTableSchemas.put(conf, profileId, tableSchema);
167 
168     OutputConf outputConf = new OutputConf(conf, profileId);
169     outputConf.writeOutputDescription(outputDesc);
170     outputConf.writeOutputTableInfo(oti);
171 
172     LOG.info("initProfile '{}' using {}", profileId, outputDesc);
173   }

175   /**
176    * Check table is not misconfigured.
177    * @param table Table to check
178    * @param outputDesc HiveOutputDescription to use
179    */
180   private static void sanityCheck(Table table,
181                                   HiveOutputDescription outputDesc) {
182     StorageDescriptor sd = table.getSd();
183     Preconditions.checkArgument(!sd.isCompressed());
184     Preconditions.checkArgument(nullOrEmpty(sd.getBucketCols()));
185     Preconditions.checkArgument(nullOrEmpty(sd.getSortCols()));
186     Preconditions.checkArgument(table.getPartitionKeysSize() ==
187         outputDesc.numPartitionValues());
188   }

200   /**
201    * Convert partition value map with ordered partition info into list of
202    * partition values.
203    * @param partitionValues Map of partition data
204    * @param fieldSchemas List of partition column definitions
205    * @return List<String> of partition values
206    */
207   private List<String> listOfPartitionValues(
208     Map<String, String> partitionValues, List<FieldSchema> fieldSchemas) {
209     List<String> values = Lists.newArrayList();
210     for (FieldSchema fieldSchema : fieldSchemas) {
211       String value = partitionValues.get(fieldSchema.getName().toLowerCase());
212       values.add(value);
213     }
214     return values;
215   }

217   @Override
218   public void checkOutputSpecs(JobContext jobContext)
219     throws IOException, InterruptedException {
220     Configuration conf = jobContext.getConfiguration();
221     OutputConf outputConf = new OutputConf(conf, myProfileId);
222 
223     HiveOutputDescription description = outputConf.readOutputDescription();
224     OutputInfo oti = outputConf.readOutputTableInfo();
225 
226     if (description == null || oti == null) {
227       LOG.error("OutputConf information is null, nothing to check");
228       return;
229     }
230 
231     if (oti.hasPartitionInfo()) {
232       if (!description.hasPartitionValues()) {
233         throw new IOException("table is partitioned but user input isn't");
234       }
235       checkPartitionDoesntExist(conf, description, oti);
236     } else {
237       if (description.hasPartitionValues()) {
238         throw new IOException("table is not partitioned but user input is");
239       } else {
240         checkTableIsEmpty(conf, description, oti);
241       }
242     }
243   }

245   /**
246    * Check if the given table is empty, that is has no files
247    * @param conf Configuration to use
248    * @param description HiveOutputDescription
249    * @param oti OutputInfo
250    * @throws IOException Hadoop Filesystem issues
251    */
252   private void checkTableIsEmpty(Configuration conf,
253     HiveOutputDescription description, OutputInfo oti)
254     throws IOException {
255     Path tablePath = new Path(oti.getTableRoot());
256     FileSystem fs = tablePath.getFileSystem(conf);
257 
258     if (fs.exists(tablePath) && FileSystems.dirHasNonHiddenFiles(fs, tablePath)) {
259       throw new IOException("Table " + description.getTableName() +
260           " has existing data");
261     }
262   }

264   /**
265    * Check that partition we will be writing to does not already exist
266    * @param conf Configuration to use
267    * @param description HiveOutputDescription
268    * @param oti OutputInfo
269    * @throws IOException Hadoop Filesystem issues
270    */
271   private void checkPartitionDoesntExist(Configuration conf,
272     HiveOutputDescription description, OutputInfo oti)
273     throws IOException {
274     HiveConf hiveConf = new HiveConf(conf, HiveApiInputFormat.class);
275     HiveMetaStoreClient client;
276     try {
277       client = new HiveMetaStoreClient(hiveConf);
278     } catch (MetaException e) {
279       throw new IOException(e);
280     }
281 
282     String db = description.getDbName();
283     String table = description.getTableName();
284 
285     if (oti.hasPartitionInfo()) {
286       Map<String, String> partitionSpec = description.getPartitionValues();
287       List<String> partitionValues = listOfPartitionValues(
288           partitionSpec, oti.getPartitionInfo());
289 
290       if (partitionExists(client, db, table, partitionValues)) {
291         throw new IOException("Table " + db + ":" + table + " partition " +
292             partitionSpec + " already exists");
293       }
294     }
295   }

297   /**
298    * Query Hive metastore if a table's partition exists already.
299    * @param client Hive client
300    * @param db Hive database name
301    * @param table Hive table name
302    * @param partitionValues list of partition values
303    * @return true if partition exists
304    */
305   private boolean partitionExists(
306       HiveMetaStoreClient client, String db, String table,
307       List<String> partitionValues) {
308     List<String> partitionNames;
309     try {
310       partitionNames = client.listPartitionNames(db, table,
311           partitionValues, (short) 1);
312       // CHECKSTYLE: stop IllegalCatch
313     } catch (Exception e) {
314       // CHECKSTYLE: resume IllegalCatch
315       return false;
316     }
317     return !partitionNames.isEmpty();
318   }

320   @Override
321   public RecordWriterImpl getRecordWriter(TaskAttemptContext taskAttemptContext)
322     throws IOException, InterruptedException {
323     HadoopUtils.setWorkOutputDir(taskAttemptContext);
324 
325     Configuration conf = taskAttemptContext.getConfiguration();
326     OutputConf outputConf = new OutputConf(conf, myProfileId);
327 
328     OutputInfo oti = outputConf.readOutputTableInfo();
329 
330     HiveUtils.setRCileNumColumns(conf, oti.getColumnInfo().size());
331     HadoopUtils.setOutputKeyWritableClass(conf, NullWritable.class);
332 
333     Serializer serializer = oti.createSerializer(conf);
334     HadoopUtils.setOutputValueWritableClass(conf,
335         serializer.getSerializedClass());
336 
337     org.apache.hadoop.mapred.OutputFormat baseOutputFormat =
338         ReflectionUtils.newInstance(oti.getOutputFormatClass(), conf);
339     // CHECKSTYLE: stop LineLength
340     org.apache.hadoop.mapred.RecordWriter<WritableComparable, Writable> baseWriter =
341         getBaseRecordWriter(taskAttemptContext, baseOutputFormat);
342     // CHECKSTYLE: resume LineLength
343 
344     StructObjectInspector soi = Inspectors.createFor(oti.getColumnInfo());
345 
346     if (!outputConf.shouldResetSlowWrites()) {
347       return new RecordWriterImpl(baseWriter, serializer, soi);
348     } else {
349       long writeTimeout = outputConf.getWriteResetTimeout();
350       return new ResettableRecordWriterImpl(baseWriter, serializer, soi, taskAttemptContext,
351           baseOutputFormat, writeTimeout);
352     }
353   }

355   /**
356    * Get the base Hadoop RecordWriter.
357    * @param taskAttemptContext TaskAttemptContext
358    * @param baseOutputFormat Hadoop OutputFormat
359    * @return RecordWriter
360    * @throws IOException Hadoop issues
361    */
362   // CHECKSTYLE: stop LineLengthCheck
363   protected static org.apache.hadoop.mapred.RecordWriter<WritableComparable, Writable> getBaseRecordWriter(
364     TaskAttemptContext taskAttemptContext,
365     org.apache.hadoop.mapred.OutputFormat baseOutputFormat) throws IOException {
366     // CHECKSTYLE: resume LineLengthCheck
367     JobConf jobConf = new JobConf(taskAttemptContext.getConfiguration());
368     int fileId = CREATED_FILES_COUNTER.incrementAndGet();
369     String name = FileOutputFormat.getUniqueName(jobConf, "part-" + fileId);
370     if (LOG.isInfoEnabled()) {
371       LOG.info("getBaseRecordWriter: Created new with file {}", name);
372     }
373     Reporter reporter = new ProgressReporter(taskAttemptContext);
374     return baseOutputFormat.getRecordWriter(null, jobConf, name, reporter);
375   }
