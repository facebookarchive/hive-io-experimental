20  import org.apache.hadoop.conf.Configuration;

22  /**
23   * Integer configuration option
24   */
25  public class IntConfOption extends AbstractConfOption {
26    /** Default value */
27    private final int defaultValue;
28  
29    /**
30     * Constructor
31     * @param key key
32     * @param defaultValue default value
33     */
34    public IntConfOption(String key, int defaultValue) {
35      super(key);
36      this.defaultValue = defaultValue;
37      AllOptions.add(this);
38    }
39  
40    /**
41     * Constructor
42     * @param key key
43     * @param defaultValue default value
44     */
45    public IntConfOption(String key, long defaultValue) {
46      super(key);
47      this.defaultValue = (int) defaultValue;
48      AllOptions.add(this);
49    }
50  
51    public int getDefaultValue() {
52      return defaultValue;
53    }
54  
55    @Override public String getDefaultValueStr() {
56      return Integer.toString(defaultValue);
57    }
58  
59    @Override public ConfOptionType getType() {
60      return ConfOptionType.INTEGER;
61    }
62  
63    /**
64     * Lookup value
65     * @param conf Configuration
66     * @return value for key, or default value if not set
67     */
68    public int get(Configuration conf) {
69      return conf.getInt(getKey(), defaultValue);
70    }
71  
72    /**
73     * Set value
74     * @param conf Configuration
75     * @param value to set
76     */
77    public void set(Configuration conf, int value) {
78      conf.setInt(getKey(), value);
79    }
80  
81    /**
82     * Set value if it's not already present
83     * @param conf Configuration
84     * @param value to set
85     */
86    public void setIfUnset(Configuration conf, int value) {
87      if (conf.get(getKey()) == null) {
88        conf.setInt(getKey(), value);
89      }
90    }
91  }
