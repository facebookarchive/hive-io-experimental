42  public class HadoopNative {
43    private static final Logger LOG = LoggerFactory.getLogger(HadoopNative.class);
44  
45    private static boolean loaded = false;
46    private static Throwable error = null;
47  
48    public static synchronized void requireHadoopNative() {
49      if (loaded) {
50        return;
51      }
52      if (error != null) {
53        throw new RuntimeException("failed to load Hadoop native library", error);
54      }
55      try {
56        loadLibrary("hadoop");
57        Field field = NativeCodeLoader.class.getDeclaredField("nativeCodeLoaded");
58        field.setAccessible(true);
59        field.set(null, true);
60  
61        // Use reflection to HACK fix caching bug in CodecPool
62        // This hack works from a concurrency perspective in this codebase
63        // because we assume that all threads that will access the CodecPool will
64        // have already been synchronized by calling requireHadoopNative() at some point.
65        field = CodecPool.class.getDeclaredField("COMPRESSOR_POOL");
66        setFinalStatic(field, new HackListMap());
67        field = CodecPool.class.getDeclaredField("DECOMPRESSOR_POOL");
68        setFinalStatic(field, new HackListMap());
69  
70        List<Class<? extends CompressionCodec>> codecs =
71            CompressionCodecFactory.getCodecClasses(new Configuration());
72        LOG.info("Compression Codecs: {}", codecs);
73  
74        loaded = true;
75      } catch (Throwable t) {
76        error = t;
77        throw new RuntimeException("failed to load Hadoop native library", error);
78      }
79    }
80  
81    private static void setFinalStatic(Field field, Object newValue)
82        throws Exception {
83      field.setAccessible(true);
84  
85      Field modifiersField = Field.class.getDeclaredField("modifiers");
86      modifiersField.setAccessible(true);
87      modifiersField.setInt(field, field.getModifiers() & ~Modifier.FINAL);
88  
89      field.set(null, newValue);
90    }
91  
92    private static void loadLibrary(String name) throws IOException {
93      URL url = Resources.getResource(HadoopNative.class, getLibraryPath(name));
94      File file = File.createTempFile(name, null);
95      file.deleteOnExit();
96      Files.copy(Resources.newInputStreamSupplier(url), file);
97      System.load(file.getAbsolutePath());
98    }
99  
100   private static String getLibraryPath(String name) {
101     return "/nativelib/" + getPlatform() + "/" + System.mapLibraryName(name);
102   }
103 
104   private static String getPlatform() {
105     String name = System.getProperty("os.name");
106     String arch = System.getProperty("os.arch");
107     return (name + "-" + arch).replace(' ', '_');
108   }

110   private static class HackListMap<K, V>
111       extends AbstractMap<K, List<? extends V>> {
112     @Override
113     public Set<Entry<K, List<? extends V>>> entrySet() {
114       return Collections.emptySet();
115     }

124       return Lists.newArrayList();
125     }
