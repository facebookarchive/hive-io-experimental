88    public static HiveWritableRecord mapToHiveRecord(MapWritable value) {
89      HiveWritableRecord record = HiveRecordFactory.newWritableRecord(HiveTools.NUM_COLUMNS);
90      for (Map.Entry<Writable, Writable> entry : value.entrySet()) {
91        IntWritable intKey = (IntWritable) entry.getKey();
92        LongWritable longValue = (LongWritable) entry.getValue();
93        record.set(intKey.get(), longValue.get());
