46  @ThreadSafe
47  class Stats {
48    public static final long MEGABYTES_FLUSH = 10;
49  
50    private static final Logger LOG = LoggerFactory.getLogger(Stats.class);

52    private final HiveStats hiveStats;

54    public Meter rowMeter;
55  
56    public Meter rawMBMeter;
57    public AtomicLong bytesTillFlush = new AtomicLong();
58  
59    public static Stats instance;
60  
61    private Stats(HiveStats hiveStats) {
62      this.hiveStats = hiveStats;
63      rowMeter = Metrics.newMeter(Stats.class, "rows", "rows", SECONDS);
64      rawMBMeter = Metrics.newMeter(Stats.class, "raw MB (estimated)", "MBs", SECONDS);
65      Metrics.newGauge(Stats.class, "used MB", new Gauge<Long>() {
66        @Override public Long value() {
67          Runtime runtime = Runtime.getRuntime();
68          return (long) ByteUnit.BYTE.toMB(runtime.totalMemory() - runtime.freeMemory());
69        }
70      });
71    }
72  
73    public static Stats get(HiveStats hiveStats) {
74      instance = new Stats(hiveStats);
75      return instance;
76    }

78    public void addRows(long numRows) {
79      rowMeter.mark(numRows);
80      double rowsFraction = numRows / (double) hiveStats.getNumRows();
81      double bytesFraction = rowsFraction * hiveStats.getRawSizeInBytes();
82      long bytes = (long) bytesFraction;
83      addBytes(bytes);
84    }
85  
86    private void addBytes(long bytes) {
87      long bytesSoFar = bytesTillFlush.addAndGet(bytes);
88      while (bytesSoFar > MB.toBytes(MEGABYTES_FLUSH)) {
89        if (bytesTillFlush.compareAndSet(bytesSoFar, 0)) {
90          rawMBMeter.mark((long) BYTE.toMB(bytesSoFar));
91        }
92        bytesSoFar = bytesTillFlush.get();
93      }
94    }

96    public static ConsoleReporter metricsReporter() {
97      return new ConsoleReporter(Metrics.defaultRegistry(), System.err, MetricPredicate.ALL);
98    }

100   public void printEnd(long timeNanos) throws IOException {
101     metricsReporter().run();
102 
103     double nsPerRow = Math.floor(timeNanos / (double) rowMeter.count());
104     double msecPerRow = NANOSECONDS.toMillis((long) nsPerRow);
105     double rawMBPerNs = hiveStats.getRawSizeInMB() / timeNanos;
106     double rawMBPerSec = SECONDS.toNanos((long) rawMBPerNs);
107     double totalMBPerNs = hiveStats.getTotalSizeInMB() / timeNanos;
108     double totalMBPerSec = BYTE.toMB(SECONDS.toNanos((long) totalMBPerNs));
109 
110     err.println("Finished.");
111     err.println("  " + rowMeter.count() + " rows");
112     err.println("  " + NANOSECONDS.toSeconds(timeNanos) + " seconds");
113     err.println("  " + nsPerRow + " ns / row");
114     err.println("  " + msecPerRow + " msec / row");
115     err.println("  " + rawMBPerNs + " raw MB / ns");
116     err.println("  " + rawMBPerSec + " raw MB / second");
117     err.println("  " + totalMBPerNs + " total MB / ns");
118     err.println("  " + totalMBPerSec + " total MB / second");
119   }

121   public void printEndBenchmark(Context context, long timeNanos,
122       OutputStream stream) throws IOException {
123     PrintWriter writer = new PrintWriter(new BufferedWriter(new OutputStreamWriter(stream)));
124     Joiner joiner = Joiner.on(",");
125 
126     long seconds = NANOSECONDS.toSeconds(timeNanos);
127     double nsPerRow = Math.floor(timeNanos / (double) rowMeter.count());
128     double msecPerRow = NANOSECONDS.toMillis((long) nsPerRow);
129     double rawMBPerNs = hiveStats.getRawSizeInMB() / timeNanos;
130     double rawMBPerSec = SECONDS.toNanos((long) rawMBPerNs);
131     double totalMBPerNs = hiveStats.getTotalSizeInMB() / timeNanos;
132     double totalMBPerSec = BYTE.toMB(SECONDS.toNanos((long) totalMBPerNs));
133 
134     writer.println(joiner.join(rowMeter.count(), rawMBMeter.count(),
135         seconds, context.opts.threads, msecPerRow, rawMBPerSec, totalMBPerSec));
136     writer.flush();
137   }
