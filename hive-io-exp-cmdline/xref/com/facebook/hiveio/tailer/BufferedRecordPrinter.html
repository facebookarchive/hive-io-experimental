18  package com.facebook.hiveio.tailer;

20  import com.facebook.hiveio.record.HiveReadableRecord;
21  
22  class BufferedRecordPrinter implements RecordPrinter {
23    @Override public void printRecord(HiveReadableRecord record, int numColumns,
24        Context context) {
25      DefaultRecordPrinter.addRecordToStringBuilder(record, numColumns, context);
26  
27      ThreadContext perThread = context.perThread.get();
28      ++perThread.recordsInBuffer;
29      if (perThread.recordsInBuffer >= context.opts.recordBufferFlush) {
30        perThread.flushBuffer();
31      }
32    }
33  }
