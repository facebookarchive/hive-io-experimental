18  package com.facebook.giraph.hive.mapreduce;

20  import org.apache.hadoop.io.MapWritable;
21  import org.apache.hadoop.io.NullWritable;
22  import org.apache.hadoop.mapreduce.InputFormat;
23  import org.apache.hadoop.mapreduce.InputSplit;
24  import org.apache.hadoop.mapreduce.JobContext;
25  import org.apache.hadoop.mapreduce.RecordReader;
26  import org.apache.hadoop.mapreduce.TaskAttemptContext;
27  
28  import com.google.common.collect.Lists;
29  
30  import java.io.IOException;
31  import java.util.List;
32  
33  public class SampleInputFormat extends InputFormat<NullWritable, MapWritable> {
34    @Override public RecordReader<NullWritable, MapWritable> createRecordReader(
35        InputSplit split, TaskAttemptContext context)
36        throws IOException, InterruptedException {
37      SampleInputSplit hSplit = (SampleInputSplit) split;
38      if (hSplit.getNum() == 0) {
39        return new SplitReader(HiveTools.mapperData1);
40      } else {
41        return new SplitReader(HiveTools.mapperData2);
42      }
43    }
44  
45    @Override public List<InputSplit> getSplits(JobContext context)
46        throws IOException, InterruptedException {
47      return Lists.<InputSplit>newArrayList(
48          new SampleInputSplit(0),
49          new SampleInputSplit(1));
50    }
51  }
